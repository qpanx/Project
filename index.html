How It Works

Detection

Five IR sensors mounted at the front constantly scan the surface contrast to differentiate between the black line and the white floor.

Processing

The Arduino reads the digital inputs from the sensors. Based on which sensor is triggered, it calculates the error correction needed.

Actuation

The L298N motor driver receives PWM signals to adjust the speed of the left and right yellow gear motors to steer the chassis.

Hardware Components

The build features a customized double-decker acrylic chassis. The wiring is managed via a breadboard on the top deck, connecting the 5-sensor array to the main controller.

main.ino 
#include <lil.abdallah1-project-1_inferencing.h>

// Sensor & Motor Pins
const int irPins[5] = {32, 33, 25, 26, 27};
int sensorValues[5];

#define IN1 14 | IN2 12 | IN3 13 | IN4 15
#define ENA 23 | ENB 22
#define LED_R 4 | LED_G 2 | LED_B 17

int motorSpeed = 110;

void setup() {
  Serial.begin(115200);
  
  for (int i = 0; i < 5; i++) {
    pinMode(irPins[i], INPUT);
  }
  
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  pinMode(ENA, OUTPUT); pinMode(ENB, OUTPUT);
  pinMode(LED_R, OUTPUT); pinMode(LED_G, OUTPUT);
  pinMode(LED_B, OUTPUT);
  
  stopMotors();
  setColor(false, false, false);
}

void loop() {
  // Read 5 IR sensors
  for (int i = 0; i < 5; i++) {
    sensorValues[i] = !digitalRead(irPins[i]);
  }

  // Prepare features for AI model
  float features[5];
  for (int i = 0; i < 5; i++) {
    features[i] = (float)sensorValues[i];
  }

  signal_t signal;
  numpy::signal_from_buffer(features, 5, &signal);

  ei_impulse_result_t result = {0};
  EI_IMPULSE_ERROR res = run_classifier(&signal, &result, false);

  // Motor decision based on AI predictions
  float forwardProb = result.classification[1].value;
  float leftProb    = result.classification[2].value;
  float rightProb   = result.classification[3].value;
  float stopProb    = result.classification[0].value;

  if (stopProb > 0.6) {
    stopMotors();
    setColor(true, true, true);
  } else if (forwardProb > 0.6) {
    moveForward();
    setColor(false, true, false); // Green
  } else if (leftProb > 0.6) {
    turnLeft();
    setColor(true, false, false); // Red
  } else if (rightProb > 0.6) {
    turnRight();
    setColor(false, false, true); // Blue
  }
  
  delay(150);
}

// Motor Control Functions
void moveForward() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
  analogWrite(ENA, motorSpeed);
  analogWrite(ENB, motorSpeed);
}

void turnLeft() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
  analogWrite(ENA, motorSpeed / 2);
  analogWrite(ENB, motorSpeed);
}

void setColor(bool r, bool g, bool b) {
  digitalWrite(LED_R, r ? LOW : HIGH);
  digitalWrite(LED_G, g ? LOW : HIGH);
  digitalWrite(LED_B, b ? LOW : HIGH);
}
